## What is ROP?
ROP (Restart Often Protocol) is a context-management discipline for multi-packet LLM workflows.
Instead of feeding every prior token back into each new prompt, ROP quantifies how much context you DROP between steps.
T = n * s + [n(n-1)/2] * s * (1-r) — where r is the ROP factor (0=full context, 1=fully stateless).
Higher ROP = leaner architecture, lower cost, better agent performance.

| ROP% | Architecture | Trade-off |
|------|--------------|-----------|
| 0%   | Naive Append | Perfect recall, quadratic cost |
| 20%  | Sliding Window | Good short-term, loses older |
| 40%  | Recursive Summarization | Balanced |
| 60%  | Vector DB / RAG | Targeted retrieval, adds latency |
| 80%  | Skeleton / State Vars | Minimal overhead, strict schema |

function renderInline(text) {
  const parts = text.split(/(\*\*[^*]+\*\*|\*[^*]+\*|`[^`]+`|\[[^\]]+\]\([^)]+\))/g);
  return parts.map((p, i) => {
    if (p.startsWith("**") && p.endsWith("**"))
      return <strong key={i} style={{ color: "#f1f5f9", fontWeight: 700 }}>{p.slice(2, -2)}</strong>;
    if (p.startsWith("*") && p.endsWith("*") && p.length > 2)
      return <em key={i} style={{ color: "#cbd5e1", fontStyle: "italic" }}>{p.slice(1, -1)}</em>;
    if (p.startsWith("`") && p.endsWith("`"))
      return (
        <code key={i} style={{
          background: "#0f1929", color: "#7dd3fc",
          padding: "1px 6px", borderRadius: 4,
          fontSize: "0.85em", fontFamily: "'JetBrains Mono', monospace",
          border: "1px solid #1e3a5f",
        }}>{p.slice(1, -1)}</code>
      );
    const link = p.match(/^\[([^\]]+)\]\(([^)]+)\)$/);
    if (link)
      return <a key={i} href={link[2]} target="_blank" rel="noopener noreferrer"
        style={{ color: "#38bdf8", textDecoration: "underline", textDecorationColor: "#1e3a5f" }}>{link[1]}</a>;
    return p;
  });
}

function parseTable(lines, startIdx) {
  // Returns { rows, headerRow, nextIdx }
  const rows = [];
  let i = startIdx;
  while (i < lines.length && lines[i].trim().startsWith("|")) {
    const row = lines[i].trim().replace(/^\||\|$/g, "").split("|").map(c => c.trim());
    rows.push(row);
    i++;
  }
  // row[1] is separator row (---|---|---)
  if (rows.length >= 2 && rows[1].every(c => /^[-:]+$/.test(c))) {
    return { headerRow: rows[0], dataRows: rows.slice(2), nextIdx: i };
  }
  return null;
}

function TableBlock({ headerRow, dataRows }) {
  return (
    <div style={{ overflowX: "auto", margin: "10px 0", borderRadius: 8, border: "1px solid #1e2d40" }}>
      <table style={{ width: "100%", borderCollapse: "collapse", fontSize: 13, minWidth: 0 }}>
        <thead>
          <tr style={{ background: "#0a1a2e" }}>
            {headerRow.map((h, i) => (
              <th key={i} style={{
                padding: "8px 12px", textAlign: "left",
                color: "#00ff88", fontWeight: 700,
                fontSize: 11, letterSpacing: "0.06em", textTransform: "uppercase",
                borderBottom: "1px solid #1e2d40",
                whiteSpace: "nowrap",
              }}>{h}</th>
            ))}
          </tr>
        </thead>
        <tbody>
          {dataRows.map((row, ri) => (
            <tr key={ri} style={{ background: ri % 2 === 0 ? "#060e1c" : "#080f1e" }}>
              {row.map((cell, ci) => (
                <td key={ci} style={{

                  borderBottom: "1px solid #0f1929",
                  fontSize: 13,

                }}>{renderInline(cell)}</td>
              ))}
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}

function MarkdownContent({ text }) {
  const lines = text.split("\n");
  const elements = [];
  let i = 0;
  let listBuffer = [];

  const flushList = () => {
    if (listBuffer.length > 0) {
      elements.push(
        <ul key={`list-${i}`} style={{ margin: "6px 0", paddingLeft: 0, listStyle: "none" }}>
          {listBuffer.map((item, li) => (
            <li key={li} style={{
              display: "flex", gap: 8, padding: "3px 0",
              color: "#94a3b8", lineHeight: 1.6, alignItems: "flex-start",
            }}>
              <span style={{ color: "#00ff88", flexShrink: 0, marginTop: 2 }}>▸</span>
              <span>{renderInline(item)}</span>
            </li>
          ))}
        </ul>
      );
      listBuffer = [];
    }
  };

  while (i < lines.length) {
    const line = lines[i];

    // Code block
    if (line.trim().startsWith("```")) {
      flushList();
      const codeLines = [];
      i++;
      while (i < lines.length && !lines[i].trim().startsWith("```")) {
        codeLines.push(lines[i]);
        i++;
      }
      elements.push(<CodeBlock key={`code-${i}`} lines={codeLines} />);
      i++;
      continue;
    }

    // Table
    if (line.trim().startsWith("|")) {
      flushList();
      const result = parseTable(lines, i);
      if (result) {
        elements.push(<TableBlock key={`table-${i}`} headerRow={result.headerRow} dataRows={result.dataRows} />);
        i = result.nextIdx;
        continue;
      }
    }

    // Headings
    if (line.startsWith("### ")) {
      flushList();
      elements.push(<h4 key={i} style={{ color: "#7dd3fc", margin: "14px 0 4px", fontSize: 13, fontWeight: 700, letterSpacing: "0.04em" }}>{renderInline(line.slice(4))}</h4>);
      i++; continue;
    }
    if (line.startsWith("## ")) {
      flushList();
      elements.push(<h3 key={i} style={{ color: "#00ff88", margin: "16px 0 5px", fontSize: 13, letterSpacing: "0.06em", textTransform: "uppercase", fontWeight: 700 }}>{renderInline(line.slice(3))}</h3>);
      i++; continue;
    }
    if (line.startsWith("# ")) {
      flushList();
      elements.push(<h2 key={i} style={{ color: "#00ff88", margin: "16px 0 6px", fontSize: 16, fontWeight: 700 }}>{renderInline(line.slice(2))}</h2>);
      i++; continue;
    }

    // List items
    if (line.match(/^[-*] /)) {
      listBuffer.push(line.slice(2));
      i++; continue;
    }

    // Numbered list
    if (line.match(/^\d+\. /)) {
      flushList();
      const num = line.match(/^(\d+)\. /)[1];
      elements.push(
        <div key={i} style={{ display: "flex", gap: 10, padding: "3px 0", color: "#94a3b8", lineHeight: 1.6 }}>
          <span style={{ color: "#00ff88", flexShrink: 0, fontWeight: 700, minWidth: 20 }}>{num}.</span>
          <span>{renderInline(line.replace(/^\d+\. /, ""))}</span>
        </div>
      );
      i++; continue;
    }

    // Horizontal rule
    if (line.match(/^---+$/)) {
      flushList();
      elements.push(<hr key={i} style={{ border: "none", borderTop: "1px solid #1e2d40", margin: "10px 0" }} />);
      i++; continue;
    }

    // Blockquote
    if (line.startsWith("> ")) {
      flushList();
      elements.push(
        <div key={i} style={{
          borderLeft: "3px solid #00ff88", paddingLeft: 12,
          color: "#64748b", margin: "6px 0", fontStyle: "italic",
        }}>{renderInline(line.slice(2))}</div>
      );
      i++; continue;
    }

    // Empty line
    if (line.trim() === "") {
      flushList();
      elements.push(<div key={i} style={{ height: 5 }} />);
      i++; continue;
    }

    // Normal paragraph
    flushList();
    elements.push(
      <div key={i} style={{ lineHeight: 1.75, color: "#cbd5e1", padding: "1px 0" }}>
        {renderInline(line)}
      </div>
    );
    i++;
  }

  flushList();
  return <>{elements}</>;
}
